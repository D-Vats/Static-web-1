            };

               Box Box1;        // Declare Box1 of type Box
               Box Box2;        // Declare Box2 of type Box

               Box1.length = 6.0; 
               Box1.breadth = 7.0;

               Box2.height = 10.0;
               Box2.length = 12.0;
               Box2.breadth = 13.0;

               volume = Box1.height * Box1.length * Box1.breadth;
